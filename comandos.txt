cd ~/ProyectoRobotica04

colcon build

# Si quieres compilar SOLO tu paquete (más rápido):

# 3. Exportar modelo del robot
export TURTLEBOT3_MODEL=burger

ros2 launch proy_andres_mundo warehouse.launch.py
